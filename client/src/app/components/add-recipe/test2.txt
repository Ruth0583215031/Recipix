import { Component, OnInit } from '@angular/core';
import { Recipe } from '../../classes/recipe';
import { FormsModule } from '@angular/forms';

@Component({
  selector: 'app-my-book',
  imports: [FormsModule],
  templateUrl: './my-book.html',
  styleUrl: './my-book.css',
})
export class MyBook implements OnInit {

  allR: Array<Recipe> = new Array<Recipe>()

  ngOnInit(): void {
    this.allR.push(new Recipe("עוגת שוקולד", "עוגת שוקולד.jfif", 30))
    this.allR.push(new Recipe("פיצה", "פיצה.jfif", 40))
    this.allR.push(new Recipe("אייס קפה ", "אייס קפה.jfif", 10))

  }

  //מבנה פונקציה
  addLike(re: Recipe) {
    re.like++
  }
  //משתנה המתקשר עם הטופס
  newR: Recipe = new Recipe()
  newEdit: boolean = true//כן =חדש
  addR() {
    if (this.newEdit)
      //הוספת המתכון החדש לרשימה
      this.allR.push(this.newR)
    else {
      //חיפוש המתכון לפי ערך ייחודי
      let i = this.allR.findIndex(x => x.nameR == this.newR.nameR)
      this.allR[i] = this.newR
      this.newEdit = true

    }
    // console.log(this.newR);
    //איפוס הטופס
    this.newR = new Recipe()

  }
  editR(r: Recipe) {
    //העתקת המידע לטופס
    this.newR = { ...r }
    //עדכון שמדובר בעריכה
    this.newEdit = false
  }
}
